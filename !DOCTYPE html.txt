    // CART MANAGEMENT WITH 'ITEM ADDED' VISUAL FEEDBACK
    function addToCart(itemId) {
      const item = catalog.find(i => i.id === itemId);
      if (!item) return;

      const qtySelect = document.getElementById(`qty-${itemId}`);
      const qty = parseInt(qtySelect ? qtySelect.value : 1);

      const existingCartIndex = cart.findIndex(c => c.id === itemId);
      if (existingCartIndex > -1) {
        cart[existingCartIndex].qty += qty;
      } else {
        cart.push({ ...item, qty, trackingStatus: 'Not Received' });
      }

      updateCartUI();

      // Show "Item Added!" button animation on card
      const addBtn = document.getElementById(`add-btn-${itemId}`);
      if (addBtn) {
        addBtn.className = "flex-1 py-1.5 bg-emerald-600 text-white text-xs font-bold rounded-lg transition-all flex items-center justify-center gap-1 shadow-inner";
        addBtn.innerHTML = `<i data-lucide="check-circle" class="w-3.5 h-3.5"></i> Item Added!`;
        lucide.createIcons();
        setTimeout(() => {
          if (addBtn) {
            addBtn.className = "flex-1 py-1.5 brand-green-bg hover:bg-emerald-900 text-white text-xs font-bold rounded-lg transition-colors flex items-center justify-center gap-1";
            addBtn.innerHTML = `<i data-lucide="plus" class="w-3.5 h-3.5"></i> Add to Order`;
            lucide.createIcons();
          }
        }, 1800);
      }

      showToast(`Item Added! (${qty}x ${item.name})`);
    }

    function showToast(message) {
      let toast = document.getElementById('toast-notification');
      if (!toast) {
        toast = document.createElement('div');
        toast.id = 'toast-notification';
        toast.className = 'fixed bottom-6 right-6 z-50 bg-emerald-900 text-white px-4 py-3 rounded-2xl shadow-xl border border-emerald-700 flex items-center gap-2.5 text-xs font-bold transition-all duration-300';

      if (window.toastTimeout) clearTimeout(window.toastTimeout);
      window.toastTimeout = setTimeout(() => {

      }, 2500);

    function updateCartQty(itemId, newQty) {
      const qty = parseInt(newQty);
      if (qty <= 0) {
        removeFromCart(itemId);
        return;
      }
      const item = cart.find(c => c.id === itemId);
      if (item) {
        item.qty = qty;
        updateCartUI();
      }
    }

    function removeFromCart(itemId) {
      cart = cart.filter(c => c.id !== itemId);
      updateCartUI();
    }

    // HELPER TO GENERATE DATE STRING IN DDMMYY FORMAT
    function getDDMMYY() {
      const now = new Date();
      const dd = String(now.getDate()).padStart(2, '0');
      const mm = String(now.getMonth() + 1).padStart(2, '0');
      const yy = String(now.getFullYear()).slice(-2);
      return `${dd}${mm}${yy}`;
    }

    // INITIAL SAMPLE DATASET (16 ORDERS FOR 15-ORDER PAGINATION DEMO)
    function generateInitialSampleOrders() {
      const ddmmyy = getDDMMYY();
      const locations = ['MILTON KEYNES', 'LONDON', 'MANCHESTER', 'BIRMINGHAM'];
      const requesters = ['SAI KISHORE (MANAGER)', 'ALIA (PROCUREMENT)', 'DAVID (STORE MGR)', 'SARAH (OPERATIONS)'];
      const statuses = ['Not Received', 'Partially Received', 'Received'];

      const initialOrders = [];
      const now = Date.now();

      for (let i = 1; i <= 16; i++) {
        const serialNum = String(17 - i).padStart(3, '0');
        const primaryRef = `TSMK-MULTI-${ddmmyy}-${serialNum}`;
        const location = locations[i % locations.length];
        const requester = requesters[i % requesters.length];
        const timestamp = now - (i * 3600000);
        const dateStr = new Date(timestamp).toLocaleDateString('en-GB', { day: '2-digit', month: 'short', year: 'numeric' }) + ', ' + new Date(timestamp).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });

        const supplierItems = [
          { ...catalog[(i * 3) % catalog.length], qty: (i % 4) + 1, trackingStatus: statuses[i % 3] },
          { ...catalog[(i * 5 + 1) % catalog.length], qty: (i % 3) + 1, trackingStatus: statuses[(i + 1) % 3] }
        ];

        const total = supplierItems.reduce((acc, item) => acc + (item.price * item.qty), 0);

        initialOrders.push({
          primaryRef,
          ddmmyy,
          serial: serialNum,
          timestamp,
          date: dateStr,
          requestedBy: requester,
          targetLocation: location,
          items: supplierItems,
          total,
          supabaseUrls: {}
        });
      }
      return initialOrders;
    }

      e.preventDefault();

      const serial = String(orders.length + 1).padStart(3, '0');
      const primaryRef = `TSMK-MULTI-${ddmmyy}-${serial}`;

      const newOrder = {
        primaryRef: primaryRef,

        date: new Date().toLocaleDateString('en-GB', { day: '2-digit', month: 'short', year: 'numeric' }) + ', ' + new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }),

        total: cart.reduce((acc, i) => acc + (i.price * i.qty), 0),

      };

      orders.unshift(newOrder);

    // ADMIN SORTING & FILTERING LOGIC
    function filterAndRenderAdminOrders() {
      const searchInput = document.getElementById('admin-search-input');
      const supplierFilter = document.getElementById('admin-supplier-filter');
      const statusFilter = document.getElementById('admin-status-filter');
      const sortSelect = document.getElementById('admin-sort-select');

      const search = searchInput ? searchInput.value.toLowerCase().trim() : '';
      const supplier = supplierFilter ? supplierFilter.value : 'ALL';
      const status = statusFilter ? statusFilter.value : 'ALL';
      const sortBy = sortSelect ? sortSelect.value : 'NEWEST';

      filteredAdminOrders = orders.filter(o => {
        const matchesSearch = !search || 
          o.primaryRef.toLowerCase().includes(search) ||
          o.requestedBy.toLowerCase().includes(search) ||
          o.targetLocation.toLowerCase().includes(search) ||
          o.items.some(i => i.name.toLowerCase().includes(search) || i.supplier.toLowerCase().includes(search));

        const matchesSupplier = supplier === 'ALL' || o.items.some(i => i.supplier === supplier);
        const matchesStatus = status === 'ALL' || o.items.some(i => (i.trackingStatus || 'Not Received') === status);

        return matchesSearch && matchesSupplier && matchesStatus;
      });

      if (sortBy === 'NEWEST') {
        filteredAdminOrders.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
      } else if (sortBy === 'OLDEST') {
        filteredAdminOrders.sort((a, b) => (a.timestamp || 0) - (b.timestamp || 0));
      } else if (sortBy === 'AMOUNT_HIGH') {
        filteredAdminOrders.sort((a, b) => b.total - a.total);
      } else if (sortBy === 'AMOUNT_LOW') {
        filteredAdminOrders.sort((a, b) => a.total - b.total);
      }

      renderAdminOrders();
    }

    // SAVE PO PDF TO SUPABASE STORAGE BUCKET (SUPABASE ONLY - NO LOCAL FALLBACK)
    async function uploadPOToSupabaseBucket(primaryRef, supplierName, pdfBlob = null) {
      const order = orders.find(o => o.primaryRef === primaryRef);
      if (!order) return;

      const items = order.items.filter(i => i.supplier === supplierName);
      const total = items.reduce((acc, i) => acc + (i.price * i.qty), 0).toFixed(2);
      const poNumber = `TSMK-${supplierName.toUpperCase().replace(/\s+/g, '')}-${order.ddmmyy}-${order.serial}`;

      if (!order.supabaseUrls) order.supabaseUrls = {};

      // Show uploading status on any matching Save to Supabase button
      const uploadBtns = document.querySelectorAll(`[onclick="uploadPOToSupabaseBucket('${primaryRef}', '${supplierName}')"]`);
      uploadBtns.forEach(btn => {
        btn.innerHTML = `<i data-lucide="loader" class="w-3 h-3 animate-spin"></i> Uploading...`;
        btn.disabled = true;
        lucide.createIcons();
      });

      if (!pdfBlob) {
        const exportContainer = document.getElementById('pdf-export-container');
        exportContainer.innerHTML = buildPOHTML(order, supplierName, items, total, poNumber);
        const element = document.getElementById('pdf-content');
        const opt = {
          margin:       [0.1, 0.2, 0.2, 0.2],
          filename:     `${poNumber}.pdf`,
          image:        { type: 'jpeg', quality: 0.98 },
          html2canvas:  { scale: 2, scrollY: 0 },
          jsPDF:        { unit: 'in', format: 'letter', orientation: 'portrait' }
        };
        try {
          pdfBlob = await html2pdf().set(opt).from(element).output('blob');
        } catch(e) {
          showToast('❌ PDF generation failed. Please try again.');
          console.error('PDF Blob creation error:', e);
          return;
        }
      }

      const filename = `${poNumber}.pdf`;

      if (!supabaseClient) {
        showToast('❌ Supabase not initialized. Check your credentials.');
        return;
      }

      try {
        const { data, error } = await supabaseClient.storage
          .from(SUPABASE_BUCKET)
          .upload(filename, pdfBlob, { contentType: 'application/pdf', upsert: true });

        if (error) {
          console.error('Supabase upload error:', error.message);
          showToast(`❌ Upload failed: ${error.message}`);
          return;
        }

        const { data: publicUrlData } = supabaseClient.storage
          .from(SUPABASE_BUCKET)
          .getPublicUrl(filename);

        if (publicUrlData && publicUrlData.publicUrl) {
          order.supabaseUrls[poNumber] = publicUrlData.publicUrl;
          filterAndRenderAdminOrders();
          showToast(`✅ PO ${poNumber} saved to Supabase!`);
        } else {
          showToast('❌ Upload succeeded but could not get public URL. Check bucket is set to Public.');
        }
      } catch (err) {
        console.error('Supabase upload exception:', err);
        showToast(`❌ Upload error: ${err.message}`);
      }
    }

    function openEditPriceModal(itemId) {

      document.getElementById('edit-item-title').textContent = item.name;

    function saveNewPrice(e) {
      e.preventDefault();
      const id = parseInt(document.getElementById('edit-item-id').value);
      const newPrice = parseFloat(document.getElementById('edit-item-price-input').value);

      const item = catalog.find(i => i.id === id);
      if (item && !isNaN(newPrice)) {
        item.price = newPrice;
        closeEditPriceModal();
        filterCatalog();
      }
    }

    // ADD SUPPLIER / INGREDIENT HANDLERS
    function handleAddSupplier(e) {
      e.preventDefault();
      const name = document.getElementById('new-supplier-name').value.trim();
      if (!name) return;

      if (!suppliersList.includes(name)) {
        suppliersList.push(name);
        populateSupplierDropdowns();
        document.getElementById('new-supplier-name').value = '';
        alert(`Supplier "${name}" added!`);
      }
    }

    function handleAddIngredient(e) {
      e.preventDefault();
      const name = document.getElementById('new-item-name').value.trim();
      const category = document.getElementById('new-item-category').value.trim();
      const size = document.getElementById('new-item-size').value.trim();
      const supplier = document.getElementById('new-item-supplier').value;
      const price = parseFloat(document.getElementById('new-item-price').value);

      const newItem = {
        id: Date.now(),
        name,
        category,
        size,
        supplier,
        price
      };

      catalog.push(newItem);
      filterCatalog();

      document.getElementById('new-item-name').value = '';
      document.getElementById('new-item-category').value = '';
      document.getElementById('new-item-size').value = '';
      document.getElementById('new-item-price').value = '';

      alert(`Ingredient "${name}" successfully added to catalog!`);
    }
